Code

import { SubNavMenu, SubNavMenuItem } from "svelte-toolkit";

<SubNavMenu type="">
  {#each items as item}
    <SubNavMenuItem>
      <a href={item.href}>{item.title}</a>
    </SubNavMenuItem>
  {/each}
</SubNavMenu>
